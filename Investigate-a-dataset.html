Research Question 4: What is the most popular genre?¶

In [125]:
# get unique genres
Genres = []
for i in df.genres:
    if type(i) == str:
        for x in i.split('|'):
            Genres.append(x)
In [126]:
GenresWithCounts = Counter(Genres)
In [127]:
